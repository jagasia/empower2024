factorial()
{
    if (( $1 <= 1 )); then
        echo 1
    else
        last=$(factorial $(( $1 - 1 )))
        echo $(( $1 * last ))
    fi
}

#include <stdio.h>

int isPrime(int num)
{
    if(num<2)
        return 0;
    if(num==2)
        return 1;
    for(int den=2;den<=num/2;den++)
    {
        if(num%den==0)
            return 0;
    }
    return 1;
}

int main()
{
    int from=0;
    int to=0;
    scanf("%d",&from);
    scanf("%d",&to);
    
    int isFirst=1;
    
    for(int i=from+1;i<to;i++)
    {
        int flag=0;
        if(isPrime(i))
        {
            //check if each digit is isPrime
            int temp=i;
            while(temp>0)
            {
                int digit=temp%10;
                
                //check if digit is isPrime
                if(!isPrime(digit))
                {
                    flag=1;
                    break;
                }
                temp/=10;
            }
            if(flag)
                continue;   
            if(!isFirst)
                printf(",");
            printf("%d",i);
            isFirst=0;
        }
    }
    return 0;
}

Task:

create a c programming that has a function named as swap 

swap function accepts 2 int arguments and swap them

note:
	when we define a function, then the arguments in function definition are formal parameters
	when we call a function, then the arguments passed to the function are actual parameters

a function swapping formal parameters, should affect the actual parameters.
For this, use pointers as formal parameters and send reference in actual parameters.

#include <stdio.h>

void swap(int*, int*);

int main()
{
 int x=0;
 int y=0;
 scanf("%d",&x);
 scanf("%d",&y);
 
 printf("After calling the swap method, look at x and y\n");
 swap(&x,&y);
 printf("%d %d",x,y);
 
}

void swap(int *a, int *b)
{
    int temp=*a;
    *a=*b;
    *b=temp;
}
