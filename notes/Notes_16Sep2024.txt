        var arr=[];
        function fn1()
        {
            var country=document.getElementById("txtCountry").value;            
            arr.push(country);
            update();
        }

        function update()
        {
            var div=document.getElementById("div1");
            div.innerHTML="";
            for(var c of arr)
            {
                div.innerHTML+=c+"<br/>";
            }
        }

        function fnRemove()
        {
            arr.pop();
            update();
        }
    </script>
</head>
<body>

<input type="button" value="Add" onclick="fn1()">    <br/>
<input type="button" value="Remove" onclick="fnRemove()"><br/>
<div id="div1"></div>
</body>
</html>



Objects
---------
	How to convert a JSON string into object



{id: '1', name: 'Gagan', department: 'CSE'}			this is object



<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script>
        function fnAdd()
        {
            var emp={};
            var id=document.getElementById("id").value;
            var name=document.getElementById("name").value;
            var department=document.getElementById("department").value;
            
            emp.id=id;
            emp.name=name;
            emp.department=department;

            console.log(emp);

            console.log(JSON.stringify(emp))
            
        }

        function fnImport()
        {
            var txt=document.getElementById("txt1").value;
            //convert json string into object
            var emp=JSON.parse(txt);
            document.getElementById("id").value=emp.id;
            document.getElementById("name").value=emp.name;
            document.getElementById("department").value=emp.department;
        }
    </script>
</head>
<body>
    <input type="text" name="" id="txt1"><br>
    <input type="button" value="Import" onclick="fnImport()">

    Id: <input type="number" name="" id="id"><br>
    Name: <input type="text" name="" id="name"><br>
    Department: <input type="text" name="" id="department"><br>
    <input type="button" value="Add" onclick="fnAdd()">


</body>
</html>
